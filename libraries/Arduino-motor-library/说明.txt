适用于AQMH3615NS直流电机驱动板

Motor(IN1，IN2，ENA);	//设置电机驱动板的接口

brake(bool);		//设置有无刹车，1表示有，0表示无，默认无

speed(int);		//设置速度，-255~255，正数正转，负数反转

bool GetBrake();		//获取刹车的设置，1表示有，0表示无

int GetSpeed();		//获取设置的速度


=============================================
IN1	IN2	ENA	OUT1、OUT2 输出 				
0	0	×	刹车 			
1	1	×	悬空 
1	0	1	全速正转 	
0	1	1	全速反转	
1	0	PWM	正转调速 		
0	1	PWM	反转调速		
=============================================
